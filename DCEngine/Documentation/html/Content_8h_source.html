172  inline void Content::AddResourceToMap(std::string & resourceName, ResourcePtr ptr, ResourceMap& map)
173  {
174  // Prevent duplicates
175  if (map.count(resourceName)) {
176  // Overwrite the current level
177  map.erase(resourceName);
178  if (DCE_TRACE_FACTORY_RESOURCE_ADD)
179  DCTrace << "Content::AddResourceToMap - " << resourceName << " is already present in the map. Overwriting. \n";
180  //return;
181  }
182 
183  map.insert(std::pair<std::string, ResourcePtr>(resourceName, ptr));
184  if (DCE_TRACE_FACTORY_RESOURCE_ADD)
185  DCTrace << "Content::AddResourceToMap - " << resourceName << " was added.\n";
186  }
