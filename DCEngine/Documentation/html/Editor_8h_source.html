250  inline bool Editor::SelectResource(std::string resourceType, ResourceMap* map, Zilch::Property * resource, ObjectPtr component, unsigned int propertyID)
251  {
252  // Get a container of all active resources
253  std::vector<const char *> resources;
254  auto resourceValue = Reflection::PropertyAsString(resource, component);
255  static int currentItem = 0;
256  for (auto resource : *map) {
257  // Push the name of it into the vector of strings
258  resources.push_back(resource.second->Name().c_str());
259  if (resource.second->getObjectName() == resourceValue)
260  currentItem = resources.size() - 1;
261  }
262 
263  // If the user selects an item...
264  if (ImGui::Combo("##propertyID", &currentItem, resources.data(), resources.size())) {
265  // Set the selected item as the current resource
266  auto selectedResource = resources.at(currentItem);
267  Zilch::ExceptionReport report;
268  Zilch::Call setCall(resource->Set, Daisy->getSystem<Reflection>()->Handler()->getState());
269  setCall.SetHandleVirtual(Zilch::Call::This, component);
270  setCall.Set(0, Zilch::String(selectedResource));
271  setCall.Invoke(report);
272  // Property was modified..
273  return true;
274  }
275  // Property was not modified
276  return false;
